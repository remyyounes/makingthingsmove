```code
<P5Wrapper sketch={(p) => {
  let rotation = 10;
  let size = 10;

  p.setup = function () {
    p.createCanvas(600, 400, p.P2D);
  };

  p.myCustomRedrawAccordingToNewPropsHandler = function (props) {
    if (props.rotation){
      rotation = props.rotation * Math.PI / 180;
    }
  };

  p.draw = function () {
    rotation += 0.05
    p.background(100);
    p.noStroke();
    p.push();
    p.translate(100, 100)
    p.rotate(rotation);
    p.rect(-size/2, -size/2, size, size);
    p.pop();
  };
}} />

```
